American Sign Language

American Sign Language is used by nearly 500,000 people to communicate with the deaf. This program teaches the ASL alphabet while demonstrating the use of arrays, random numbers, and functions. Use the A button to select an answer, use the B button to change the selection. After 3 incorrect answers the quiz is over. Note this is a large program and may take a bit to load in the simulator.

input.buttonA.onEvent(ButtonEvent.Down, function () {
    if (selectionIcon.x == 9 && correctAnswerPosition == 1) {
        generateRandomLetter()
        info.changeScoreBy(1)
        music.playTone(587, music.beat(BeatFraction.Half))
    } else if (selectionIcon.x == 31 && correctAnswerPosition == 2) {
        generateRandomLetter()
        info.changeScoreBy(1)
        music.playTone(587, music.beat(BeatFraction.Half))
    } else if (selectionIcon.x == 89 && correctAnswerPosition == 3) {
        generateRandomLetter()
        info.changeScoreBy(1)
        music.playTone(587, music.beat(BeatFraction.Half))
    } else if (selectionIcon.x == 114 && correctAnswerPosition == 4) {
        generateRandomLetter()
        info.changeScoreBy(1)
        music.playTone(587, music.beat(BeatFraction.Half))
    } else {
        info.changeLifeBy(-1)
        music.playTone(147, music.beat(BeatFraction.Half))
    }
})

function generateUniqueAnswers () {
    position1RandomNumber = randint(1, 25)
    if (position1RandomNumber == indexNumber) {
        position1RandomNumber = randint(1, 25)
    }
    position2RandomNumber = randint(1, 25)
    if (position2RandomNumber == indexNumber || position1RandomNumber == position2RandomNumber) {
        position2RandomNumber = randint(1, 25)
    }
    position3RandomNumber = randint(1, 25)
    if (position3RandomNumber == indexNumber || position2RandomNumber == position3RandomNumber) {
        position3RandomNumber = randint(1, 25)
    }
    position4RandomNumber = randint(1, 25)
    if (position4RandomNumber == indexNumber || position3RandomNumber == position4RandomNumber) {
        position4RandomNumber = randint(1, 25)
    }
}
function generateCorrectAnswerPosition () {
    correctAnswerPosition = randint(1, 4)
}
