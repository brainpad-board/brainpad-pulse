        pins.P1.digitalWrite(true)
    } else if (seconds == 0 && (minutes > 0 && hours >= 0)) {

        seconds += -1
    }
    if (mode == 2) {

            seconds = 0

        if (minutes >= 59) {
            minutes = 0
            hours += 1
        }
        if (hours > 12) {
            hours = 1
        }
        seconds += 1
    }
